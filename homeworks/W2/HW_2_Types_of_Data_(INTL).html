Task 2

Given a dictionary of students with their scores, find the average score of each student across all subjects and identify the student with the highest average score.

In[6]:
students_grades = {
    "John": [85, 90, 78],
    "Alice": [88, 92, 80],
    "Bob": [75, 85, 72],
    "Diana": [90, 95, 94],
    "Charlie": [70, 65, 80]
}
In[7]:
# Calculate and print the average grade for each student
student_averages = {}
for student, grades in students_grades.items():
    average_grade = sum(grades) / len(grades)
    student_averages[student] = average_grade
    print(f"{student}: Average Grade = {average_grade:.2f}")

John: Average Grade = 84.33
Alice: Average Grade = 86.67
Bob: Average Grade = 77.33
Diana: Average Grade = 93.00
Charlie: Average Grade = 71.67
In[8]:
# Find the student with the highest average grade
highest_avg_student = None
highest_average = -1

for student, average in student_averages.items():
    if average > highest_average:
        highest_average = average
        highest_avg_student = student

print(f"Student with the highest average score: {highest_avg_student} (Average: {highest_average:.2f})")
